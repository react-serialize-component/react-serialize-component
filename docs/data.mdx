# 数据源模块

```json schema=true
{
  "type": "tpl",
  "DataSource": {
    "page": {
      "data": { "title": "test" },
      "list": "http://jianxcao.test/test1"
    }
  },
  "onInit": "${page.fetchList()}",
  "bindData": {
    "page": true
  },
  "tpl": "<pre>${page|json}</pre>"
}
```

```json schema=true
{
  "type": "tpl",
  "DataSource": {
    "page": {
      "data": { "title": "test" },
      "list": {
        "method": "post",
        "url": "http://jianxcao.test/test1",
        "params": {
          "a": 1,
          "b": 2
        }
      }
    }
  },
  "onInit": "${page.fetchList()}",
  "bindData": {
    "page": true
  },
  "tpl": "<pre>${page|json}</pre>"
}
```

```json schema=true
{
  "type": "page",
  "bindData": {
    "page": true
  },
  "content": {
    "$type": "button",
    "type": "default",
    "onClick": "${alert('test')}",
    "children": "test button"
  }
}
```
